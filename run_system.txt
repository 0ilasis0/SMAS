@echo off
:: 切換到目前批次檔所在的目錄
cd /d "%~dp0"

:: 1. 啟動背景的 Streamlit 伺服器 (使用 Portable Python)
:: headless=true 確保它不會自己亂彈出瀏覽器分頁
start /b .\python\python.exe -m streamlit run src\app.py --server.port 8501 --server.headless true

:: 2. 等待 5 秒鐘，讓 AI 模型與 Streamlit 有時間預熱啟動
timeout /t 5 /nobreak > nul

:: 3. 啟動 Chrome App Mode 偽裝成桌面應用程式
:: 這會隱藏網址列、分頁與書籤，看起來就像原生的 Windows 軟體
start chrome --app=http://localhost:8501
